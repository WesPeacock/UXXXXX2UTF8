sub u2txt{
my ($txtstring) = @_;

return $txtstring if $txtstring !~ m/U\+/;
my $vstring = "";
my @num =  split (/U\+/, $txtstring);
for my $n (@num) {
   next if !$n; #skip first null
   $vstring = $vstring . chr(hex($n));
   }
return $vstring;
};
